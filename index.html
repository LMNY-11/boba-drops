using namespace std;
const int N = 2e5+5;
const int mod = 1e9+7;
int st[4N];
int lz_L[4N];
int lz_I[4N];
int a[N];
int n,q;
int sum(int i){
    return i(i+1)/2;
}
void apply(int t,int l,int r,int L,int I){
    st[t] += L(r-l+1) + I(l*(r-l+1) + sum(r-l));
    lz_L[t]+= L;
    lz_I[t]+= I;
}
void push(int t,int l,int r){
    int mid=(l+r)>>1;
    if(lz_I[t]!=0  lz_L[t]!=0){
        apply(t<<1,l,mid,lz_L[t],lz_I[t]);
        apply((t<<1)|1,mid+1,r,lz_L[t],lz_I[t]);
        lz_L[t]=0;
        lz_I[t]=0;
    }
}
void update(int t,int l,int r,int u,int v,int Lx){
    if(l>vr>1;
    push(t,l,r);
    update(t<<1,l,mid,u,v,Lx);
    update((t<<1)|1,mid+1,r,u,v,Lx);
    st[t] = (st[t<<1]+st[(t<<1)|1]);
}
int get(int t,int l,int r,int u,int v){
    if(l>v||r>1;
    push(t,l,r);
    return (get(t<<1,l,mid,u,v) + get((t<<1)|1,mid+1,r,u,v));
}
void build(int t,int l,int r){
    if(l==r){
        st[t] = a[l];
        return;
    }
    int mid=(l+r)>>1;
    build(t<<1,l,mid);
    build((t<<1)|1,mid+1,r);
    st[t] = (st[t<<1] + st[(t<<1)|1]);
}

    ios_base::sync_with_stdio(false);

    cin>>n>>q;
    FOR(i,1,n)
        cin>>a[i];
    build(1,1,n);
    while(q--){
        int type,u,v;
        cin>>type>>u>>v;
        if(type==1){
            update(1,1,n,u,v,u);
        }

            cout<<get(1,1,n,u,v)<<"\n";
        }
    }

}
